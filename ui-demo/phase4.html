SDK Integration Guide

Embed verification into your application using our SDK, API, or web component.

Quick Start

// Install: npm install @vc-infra/verify-sdk

import { VerifySDK } from '@vc-infra/verify-sdk';

const verifier = new VerifySDK({
  apiKey: 'your-api-key',
  trustedIssuers: ['did:web:uwi.edu:issuer'],
});

// Verify a credential
const result = await verifier.verify(credentialJSON);

console.log(result.valid);        // true
console.log(result.checks);       // { signature: 'pass', ... }
console.log(result.claims);       // { name: 'Jane Doe', ... }
console.log(result.latencyMs);    // 280
